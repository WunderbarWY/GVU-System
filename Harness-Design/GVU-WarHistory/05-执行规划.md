applyIncremental: function(issues) {
  var self = this;
  issues.forEach(function(issue) {
    var localIssue = G.linearIssues[issue.id];
    var wasDone = localIssue && localIssue.state === 'done';
    var isDone = issue.state === 'done';

recordKill: function(data) {

## Phase 3: 幂等性 + 离线队列

### 文件：`app.js`（`WarHistoryStore`）

#### 3.1 改造 `_syncRecord` 为幂等 upsert

```js
_syncRecord: function(record) {
  var self = this;
  if (!G.user || !G.user.id) {
    self._enqueue(record);
    return;
  }

  var payload = {
    client_id: record.clientId,
    user_id: G.user.id,
    type: record.type,
    mission_id: record.missionId || null,
    ship_name: record.shipName || null,
    time: record.time,
    description: record.desc || null,
    metadata: record.metadata || {}
  };

  supabaseApi.upsert('war_history', payload, { onConflict: 'client_id' })
    .then(function() {
      record.synced = true;
      self.save();
      self._dequeue(record.clientId);
    })
    .catch(function(err) {
      console.warn('war_history sync failed', err);
      self._enqueue(record);
    });
}
```

若 `supabaseApi.upsert` 不存在，在 `src/api/supabase.js` 补充：

```js
upsert: function(table, payload, options) {
  options = options || {};
  var q = this.client.from(table).upsert(payload);
  if (options.onConflict) q = q.onConflict(options.onConflict);
  return q;
}
```

**预期行为**：
- 同一 `client_id` 重复 upsert 不会创建重复行。
- 网络错误时记录进入队列。

#### 3.2 新增离线队列

```js
_queueKey: 'gvu_war_history_queue',

_enqueue: function(record) {
  var queue = this._loadQueue();
  var exists = queue.some(function(r) { return r.clientId === record.clientId; });
  if (!exists) {
    queue.push(record);
    this._saveQueue(queue);
  }
},

_dequeue: function(clientId) {
  var queue = this._loadQueue().filter(function(r) {
    return r.clientId !== clientId;
  });
  this._saveQueue(queue);
},

_loadQueue: function() {
  try {
    return JSON.parse(localStorage.getItem(this._queueKey)) || [];
  } catch (e) {
    return [];
  }
},

_saveQueue: function(queue) {
  localStorage.setItem(this._queueKey, JSON.stringify(queue));
},

flushQueue: function() {
  var self = this;
  this._loadQueue().forEach(function(record) {
    self._syncRecord(record);
  });
}
```

在应用启动或网络恢复时调用 `flushQueue`：

```js
window.addEventListener('online', function() {
  WarHistoryStore.flushQueue();
});

window.addEventListener('DOMContentLoaded', function() {
  WarHistoryStore.flushQueue();
});
```

**预期行为**：
- 离线记录保存在 `localStorage`。
- 网络恢复后自动同步队列。
- 同步成功后从队列移除。

describe('WarHistoryStore', function() {
